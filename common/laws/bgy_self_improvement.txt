	can_enact = {
		exists = ig:ig_burgundists
		ig:ig_burgundists = {
			is_in_government = yes
		}
		OR = {
			has_law = law_type:law_autocracy
			has_law = law_type:law_technocracy
			has_law = law_type:law_oligarchy
			has_law = law_type:law_single_party_state
		}
	}

	pop_support = {
		value = 0
		# Burgundians will support this
		add = {
			desc = "POP_BURGUNDIAN"
			if = {
				limit = {
					culture = cu:burgundian
				}
				value = 250
			}
		}
		# Spartanists will support this
		add = {
			desc = "POP_SPARTANIST"
			if = {
				limit = {
					religion = rel:spartanist
				}
				value = 250
			}
		}
	}
	
	# AI shouldn't try to abolish institutions

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = {
				ruler = {
					has_ideology = ideology:ideology_bgy_burgundist_leader
				}
			}
			add = 50
		}
		if = {
			limit = {
				exists = ig:ig_burgundists

					is_in_government = yes
				}
			}
			add = 50
		}
		if = {
			limit = {
				exists = ig:ig_burgundists
				ig:ig_burgundists = {
					is_marginal = no
				}
			}
			add = 10
		}
		if = {
			limit = {
				exists = ig:ig_burgundists
				ig:ig_burgundists = {
					is_marginal = yes
				}
			}
			multiply = 0
		}
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}
