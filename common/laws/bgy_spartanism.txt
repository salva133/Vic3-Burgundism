law_bgy_spartanism = {
	group = lawgroup_labor_rights
	icon = "gfx/interface/icons/law_icons/spartanism.dds"
	
	progressiveness = 0
	
	unlocking_technologies = { totalism }

	can_enact = {
		exists = ig:ig_burgundists
		ig:ig_burgundists = {
			is_in_government = yes
		}
		# OR = {
		# 	has_law = law_type:law_autocracy
		# 	has_law = law_type:law_technocracy
		# 	has_law = law_type:law_oligarchy
		# 	has_law = law_type:law_single_party_state
		# }
		country_has_voting_franchise = no
		has_law = law_type:law_no_social_security
		has_law = law_type:law_no_womens_rights
	}

	institution = institution_minilab

	modifier = {
		country_institution_minilab_max_investment_add = 5
	}
	
	on_enact = {
		recalculate_pop_ig_support = yes
	}
	
	pop_support = {
		value = 0
		# Burgundians will support this
		add = {
			desc = "POP_BURGUNDIAN"
			if = {
				limit = {
					has_pop_culture = cu:burgundian
				}
				value = 250
			}
		}
		# Spartanists will support this
		add = {
			desc = "POP_SPARTANIST"
			if = {
				limit = {
					has_pop_religion = spartanist
				}
				value = 250
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		if = {
			limit = {
				ruler = {
					has_ideology = ideology:ideology_bgy_burgundist_leader
				}
			}
			add = 50
		}
		if = {
			limit = {
				exists = ig:ig_burgundists
				ig:ig_burgundists = {
					is_in_government = yes
				}
			}
			add = 50
		}
		if = {
			limit = {
				exists = ig:ig_burgundists
				ig:ig_burgundists = {
					is_marginal = no
				}
			}
			add = 10
		}
		if = {
			limit = {
				exists = ig:ig_burgundists
				ig:ig_burgundists = {
					is_marginal = yes
				}
			}
			multiply = 0
		}
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}
