### Seventh Seal broken
judgement_day.7 = {
	type = country_event
	placement = root
	title = judgement_day.7.t
	desc = judgement_day.7.d
	flavor = judgement_day.7.f

	duration = 3

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
		NOT = { has_global_var = judgement_day_completed_glob_var }
	}

	immediate = {
		set_global_variable = judgement_day_completed_glob_var
	}

	option = {
		name = judgement_day.7.a 
		default_option = yes

		every_state = {
			limit = { NOT = { owner = root } }
			state_region = {
				add_devastation = 99
			}
			add_modifier = {
				name = end_time_mortality
				months = normal_modifier_time
			}
		}
	}
}
